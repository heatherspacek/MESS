# address: 8006D1EC
# the original hooks the top of PlayerThink_Damage for some reason
# may as well do the same

.set BUTTONS, 0x80479C3F
.set PLAYER_BLOCK_BASE, 0x80453080
.set PLAYER_BLOCK_SIZE, 0xE90
.set PLAYER_BLOCK_SLOT_TYPE, 0x8
.set SLOT_TYPE_NA, 0x3
.set PLAYER_BLOCK_DAMAGE, 0x60
.set PLAYER_BLOCK_GOBJ, 0xB0
.set GOBJ_USERDATA, 0x2C
.set FIGHTER_DAMAGE, 0x1830

# r4: pointer to floats.
# r5: pointer to buttons released this frame.
# r6: buttons released this frame.
# r7: integer damage delta. original code uses a word instead of a short.
# f0: float damage delta. original code uses a gpr and converts but we'll just load from r4 instead.

mflr r5
bl code_start
.float -1.0
.float 1.0
.float -10.0
.float 10.0
.float 0.0
code_start:
mflr r4
mtlr r5

# read buttons released this frame
lis r5, BUTTONS@h
ori r5, r5, BUTTONS@l
lbz r6, 0(r5)

cmpwi r6, 1
beq neg1
cmpwi r6, 2
beq pos1
cmpwi r6, 4
beq neg10
cmpwi r6, 8
beq pos10
b exit

neg1:
li r7, -1
lfs f0, 0x0(r4)
b continue

pos1:
li r7, 1
lfs f0, 0x4(r4)
b continue

neg10:
li r7, -10
lfs f0, 0x8(r4)
b continue

pos10:
li r7, 10
lfs f0, 0xC(r4)
continue:



# r5: pointer to buttons released this frame.
# r7: integer damage delta.
# f0: float damage delta.
# registers avaible: r4, r6, r8+


# we've gotta clear the "buttons released this frame" because this code runs multiple times a frame
# (because of the hook location, it runs once per fighter)

li r6, 0
stb r6, 0(r5)


# setting up the loop
# r4: loop index
# r6: pointer to Player
# r8: pointer to Fighter
# f2: 0.0f

# have to do this before we overwrite r4
lfs f2, 0x10(r4)

li r4, 0

lis r6, PLAYER_BLOCK_BASE@h
ori r6, r6, PLAYER_BLOCK_BASE@l

loop_start:

# check if slot is in use
lwz r8, PLAYER_BLOCK_SLOT_TYPE(r6)
cmpwi r8, SLOT_TYPE_NA
beq advance_loop

# grab Fighter pointer
lwz r8, PLAYER_BLOCK_GOBJ(r6)
lwz r8, GOBJ_USERDATA(r8)

# modify Player damage
lhz r9, PLAYER_BLOCK_DAMAGE(r6)
add r9, r9, r7
# if it would be negative, write 0 instead
cmpwi r9, 0
bge modify_player
li r9, 0
modify_player:
sth r9, PLAYER_BLOCK_DAMAGE(r6)

# modify Fighter damage
lfs f1, FIGHTER_DAMAGE(r8)
fadd f1, f1, f0
# if it would be negative, write 0 instead
fsel f1, f1, f1, f2 # f1 = (f1 >= 0) ? f1 : f2;
stfs f1, FIGHTER_DAMAGE(r8)

# go next
advance_loop:
addi r4, r4, 1
addi r6, r6, PLAYER_BLOCK_SIZE
cmpwi r4, 6
blt loop_start

# all done! original instruction:
exit:
mflr r0
